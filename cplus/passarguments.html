Pass arguments to function¶

take value as parameter¶

void mySwap(int a, int b) {
      int tmp = a;
      a = b;
      b = tmp;
}

take pointer as parameter¶

void mySwap2(int* a, int* b) {
      int tmp = *a;
      *a = *b;
      *b = tmp;

}

take reference as parameter¶

void mySwap3(int& a, int& b) {
      int tmp = b;
      b = a;
      a = tmp;
}
