1. -fverbose-asm 自行注释对应的C代码¶

source code

#include <iostream>
#include <string>

using namespace std;

template <typename T> T Sub(T tv1, T tv2) { return tv1 - tv2; }

int main(int arch, char** argv)
{
     int subv = Sub(3, 5);

     cout << "subv = " << subv << endl;

     double subv2= Sub(4.7 , 2.1);

     cout << "subv2 = " << subv2 << endl;

     return 0;
}

添加-fverbose-asm

g++ -S demo.c -fverbose-asm //编译出的demo.s文件中会自行注释对应的C代码。
